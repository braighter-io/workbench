**Goal:** Ship `db-button` — the first foundational interactive primitive in the `db-*` brick set — as a standalone attribute directive on native `<button>`/`<a>` with `variant`/`size`/`loading`, tokens-only CSS, and a pure-function-tested core.

**Architecture:** Three-part split mirroring `<db-pitch>`: a pure decision module (`button-classes.ts`, the only unit-tested part), a thin reactive directive that binds it to host attributes (`button.directive.ts`), and a tokens-only stylesheet in the CSS lib (`design-system-css/.../button.css`). All logic lives in the pure module so the flaky TestBed harness is sidestepped entirely.

**Tech Stack:** Angular standalone directive (signal `input()`, `computed()`, host bindings), CSS custom properties (design-system tokens), vitest (pure-function tests), Nx (`build`/`test`/`lint`).

```ts
// Pure decision logic for the DbButton directive. No Angular import — this is
// the unit-tested core (mirrors pitch-geometry.ts), kept free of the TestBed.

export type DbButtonVariant = 'primary' | 'secondary' | 'ghost' | 'danger';
export type DbButtonSize = 'sm' | 'md' | 'lg';

export interface ButtonHost {
  /** Space-joined class list for the host `[class]` binding. */
  classes: string;
  /** `aria-busy` attribute value, or `null` to omit the attribute. */
  ariaBusy: 'true' | null;
}

/** Resolve the host class list + aria-busy for a button's current state. */
export function resolveButtonHost(
  variant: DbButtonVariant,
  size: DbButtonSize,
  loading: boolean,
): ButtonHost {
  const classes = ['db-btn', `db-btn--${variant}`, `db-btn--${size}`];
  if (loading) classes.push('is-loading');
  return { classes: classes.join(' '), ariaBusy: loading ? 'true' : null };
}

/** Whether the directive should cancel activation (pointer + keyboard) now. */
export function shouldBlockActivation(loading: boolean): boolean {
  return loading;
}
```

```ts
import {
  Directive,
  ElementRef,
  booleanAttribute,
  computed,
  inject,
  input,
} from '@angular/core';
import {
  resolveButtonHost,
  shouldBlockActivation,
  type DbButtonSize,
  type DbButtonVariant,
} from './button-classes.js';

export type { DbButtonSize, DbButtonVariant } from './button-classes.js';

/**
 * `db-button` — design-system styling for a native button or link.
 *
 * Apply to a `<button>` or `<a>`; native semantics (type, disabled, focus,
 * keyboard activation, role) are inherited. `loading` shows a CSS spinner and
 * guards activation (pointer + keyboard). The directive never overrides the
 * native `type` default — set `type="button"` yourself inside a form if needed.
 *
 * Requires the companion stylesheet:
 *   `@import '@de-braighter/design-system-css/components/button.css';`
 *
 * @maturity experimental
 * @since 1.6.0
 */
@Directive({
  selector: 'button[db-button], a[db-button]',
  standalone: true,
  host: {
    '[class]': 'hostState().classes',
    '[attr.aria-busy]': 'hostState().ariaBusy',
    '[attr.aria-disabled]': 'ariaDisabled()',
    '(click)': 'onClick($event)',
  },
})
export class DbButton {
  private readonly el = inject(ElementRef);
  /** tagName is stable for the element's lifetime — read once. */
  private readonly isAnchor = (this.el.nativeElement as HTMLElement).tagName === 'A';

  readonly variant = input<DbButtonVariant>('primary');
  readonly size = input<DbButtonSize>('md');
  readonly loading = input(false, { transform: booleanAttribute });

  protected readonly hostState = computed(() =>
    resolveButtonHost(this.variant(), this.size(), this.loading()),
  );

  /** Anchors have no native `disabled`; reflect it while loading. */
  protected readonly ariaDisabled = computed(() =>
    this.isAnchor && this.loading() ? 'true' : null,
  );

  protected onClick(event: Event): void {
    if (shouldBlockActivation(this.loading())) {
      event.preventDefault();
      event.stopImmediatePropagation();
    }
  }
}
```
